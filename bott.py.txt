import os
import requests
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes

TOKEN = os.getenv("BOT_TOKEN")
API_KEY = os.getenv("API_KEY")

headers = {"x-apisports-key": API_KEY}


# 🌍 قاعدة فرق كبيرة (VIP core)
teams_db = {
    "ريال مدريد": "Real Madrid",
    "برشلونة": "Barcelona",
    "مانشستر سيتي": "Manchester City",
    "ليفربول": "Liverpool",
    "أرسنال": "Arsenal",
    "تشيلسي": "Chelsea",
    "بايرن ميونخ": "Bayern Munich",
    "باريس سان جيرمان": "Paris Saint Germain",
    "ميلان": "AC Milan",
    "إنتر": "Inter",
    "يوفنتوس": "Juventus",
}


# 🔍 مطابقة ذكية
def resolve(name):
    name = name.strip().lower()

    for ar, en in teams_db.items():
        if name == ar.lower():
            return en

    for ar, en in teams_db.items():
        if name in ar.lower() or ar.lower() in name:
            return en

    for en in teams_db.values():
        if name in en.lower():
            return en

    return None


# 📊 آخر مباريات
def last(team):
    url = f"https://v3.football.api-sports.io/fixtures?team={team}&last=5"
    return requests.get(url, headers=headers).json()


# 🧠 VIP Rating system
def rating(data):
    wins = 0
    goals_for = 0
    goals_against = 0
    games = 0

    for m in data.get("response", []):
        games += 1

        home = m["teams"]["home"]
        away = m["teams"]["away"]

        if home["winner"] or away["winner"]:
            wins += 1

        goals_for += m["goals"]["home"] + m["goals"]["away"]
        goals_against += 1

    if games == 0:
        return 0, 0, 0

    score = (wins * 4) + (goals_for * 0.6) - (goals_against * 0.3)
    return score, goals_for, goals_against


# 🎯 تحليل VIP
def analyze(t1, t2):
    d1 = last(t1)
    d2 = last(t2)

    if not d1.get("response") or not d2.get("response"):
        return "❌ لا توجد بيانات كافية"

    r1, g1, a1 = rating(d1)
    r2, g2, a2 = rating(d2)

    text = f"🔥 VIP ANALYSIS 🔥\n{t1} vs {t2}\n\n"

    # 🏆 الفائز
    if r1 > r2:
        winner = t1
        text += f"🏆 Winner: {t1}\n"
    elif r2 > r1:
        winner = t2
        text += f"🏆 Winner: {t2}\n"
    else:
        winner = "DRAW"
        text += "🤝 Draw Match\n"

    # ⚽ Goals
    avg = (g1 + g2) / 10
    over = avg > 2.5
    text += "🔥 Over 2.5\n" if over else "⚽ Under 2.5\n"

    # 🎯 BTTS
    btts = g1 > 6 and g2 > 6
    text += "🎯 BTTS YES\n" if btts else "🚫 BTTS NO\n"

    # 📊 Confidence
    confidence = min(97, max(60, int((r1 + r2) * 4)))
    text += f"\n📈 Confidence: {confidence}%"

    # ⚠️ Match difficulty
    diff = abs(r1 - r2)
    if diff < 5:
        level = "🟡 MEDIUM"
    elif diff < 10:
        level = "🔴 HARD"
    else:
        level = "🟢 EASY"

    text += f"\n⚠️ Difficulty: {level}"

    # 🎯 SAFE BET (VIP FEATURE)
    if winner != "DRAW":
        if over and btts:
            safe = "BTTS + Over 2.5"
        elif over:
            safe = "Over 2.5 Goals"
        else:
            safe = f"{winner} Win"
    else:
        safe = "No Safe Bet"

    text += f"\n\n🎯 SAFE BET: {safe}"

    return text


# ▶️ start
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "🔥 VIP BOT READY\n"
        "اكتب: ريال مدريد ضد برشلونة"
    )


# 📩 handle
async def handle(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text

    if "ضد" in text:
        t1, t2 = text.split("ضد")
    elif "vs" in text.lower():
        t1, t2 = text.lower().split("vs")
    else:
        await update.message.reply_text("❌ اكتب: فريق ضد فريق")
        return

    team1 = resolve(t1)
    team2 = resolve(t2)

    if not team1 or not team2:
        await update.message.reply_text(
            "❌ لم يتم التعرف على الفريق\n"
            "جرّب كتابة الاسم بشكل أوضح"
        )
        return

    result = analyze(team1, team2)
    await update.message.reply_text(result)


# 🚀 تشغيل
app = Application.builder().token(TOKEN).build()

app.add_handler(CommandHandler("start", start))
app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle))

print("🔥 VIP BOT RUNNING")
app.run_polling()
